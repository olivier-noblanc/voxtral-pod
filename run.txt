const CHUNK_SIZE = 4 * 1024 * 1024; // 4 MB

let getClientId = () => {
    let id = localStorage.getItem("sota_client_id");
    if (!id) { id = "user_" + crypto.randomUUID().slice(0, 8); localStorage.setItem("sota_client_id", id); }
    return id;
};

async function uploadFile() {
    const fileInput = document.getElementById('audioFile');
    if (!fileInput.files.length) return alert("Veuillez sélectionner un fichier.");

    const file = fileInput.files[0];
    const totalChunks = Math.ceil(file.size / CHUNK_SIZE);
    const fileId = crypto.randomUUID();
    const clientId = getClientId();

    const progress = document.getElementById('uploadProgress');
    const status   = document.getElementById('batchStatus');
    const btn      = document.getElementById('uploadBtn');

    progress.style.display = 'block';
    progress.value = 0;
    btn.setAttribute('aria-busy', 'true');
    status.innerText = "Démarrage de l'upload...";

    for (let i = 0; i < totalChunks; i++) {
        const start = i * CHUNK_SIZE;
        const end   = Math.min(start + CHUNK_SIZE, file.size);
        const chunk = file.slice(start, end);

        const formData = new FormData();
        formData.append("file_id", fileId);
        formData.append("client_id", clientId);
        formData.append("chunk_index", String(i));
        formData.append("total_chunks", String(totalChunks));
        formData.append("file", chunk, file.name);

        status.innerText = `Upload : chunk ${i + 1}/${totalChunks}…`;

        let res;
        try {
            res = await fetch('/batch_chunk', { method: 'POST', body: formData });
        } catch (networkErr) {
            status.innerText = `❌ Erreur réseau au chunk ${i + 1}: ${networkErr.message}`;
            btn.removeAttribute('aria-busy');
            return;
        }

        if (!res.ok) {
            const body = await res.text();
            status.innerText = `❌ Erreur serveur ${res.status} au chunk ${i + 1}: ${body}`;
            btn.removeAttribute('aria-busy');
            return;
        }

        progress.value = ((i + 1) / totalChunks) * 100;
    }

    status.innerText = "Fichier réassemblé — Inférence GPU en cours…";
    pollStatus(fileId, btn, status);
}

async function pollStatus(fileId, btn, status) {
    const interval = setInterval(async () => {
        try {
            const res  = await fetch(`/status/${fileId}`);
            const data = await res.json();

            if (data.status === "done") {
                clearInterval(interval);
                btn.removeAttribute('aria-busy');
                status.innerText = "✅ Transcription terminée.";
                const box = document.getElementById('batchResult');
                box.style.display = 'block';
                box.innerText = JSON.stringify(data.result, null, 2);
            } else if (data.status === "error") {
                clearInterval(interval);
                btn.removeAttribute('aria-busy');
                status.innerText = `❌ ERREUR GPU : ${data.error}`;
            } else if (data.status && typeof data.status === "string" && data.status.startsWith("processing:")) {
                // Affichage de la progression dynamique renvoyée par le backend
                let progressText = data.status.replace("processing:", "").trim();
                status.innerText = `⏳ Inférence GPU : ${progressText}`;
            }
        } catch (e) {
            // Retry silencieux
        }
    }, 2000);
}

                class AudioProcessor extends AudioWorkletProcessor {
                    constructor() {
                        super();
                        this.buffer = new Float32Array(1600); // 100ms @ 16kHz
                        this.pos = 0;
                    }
                    process(inputs) {
                        const input = inputs[0][0];
                        if (!input) return true;
                        
                        for (let i = 0; i < input.length; i++) {
                            this.buffer[this.pos++] = input[i];
                            if (this.pos >= 1600) {
                                this.port.postMessage(this.buffer);
                                this.pos = 0;
                            }
                        }
                        return true;
                    }

            const blob = new Blob([processorCode], { type: 'application/javascript' });

            await audioContext.audioWorklet.addModule(blobUrl);
            URL.revokeObjectURL(blobUrl);

            const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
            source    = audioContext.createMediaStreamSource(stream);
            processor = new AudioWorkletNode(audioContext, 'audio-processor');

            source.connect(processor);
            // NE PAS connecter processor à destination pour éviter le feedback

            processor.port.onmessage = (e) => {
                const inputData = e.data; // Float32Array

                // Visualisation RMS
                let sum = 0;
                for (let i = 0; i < inputData.length; i++) sum += inputData[i] ** 2;

class SotaASR:
    def __init__(self, model_id: str):
        self.model_id = model_id

        free_vram_gb = 0.0
        if torch.cuda.is_available():
            free_vram, total_vram = torch.cuda.mem_get_info()
            free_vram_gb = free_vram / (1024 ** 3)
            print(f"[*] VRAM : Libre={free_vram_gb:.2f} GB / Total={total_vram / (1024**3):.2f} GB")

        print(f"[*] Chargement du modèle '{model_id}' sur {device} ({torch_dtype})…")

        if model_id == "voxtral":
            from transformers import VoxtralRealtimeForConditionalGeneration, AutoProcessor, BitsAndBytesConfig
            repo_id = "mistralai/Voxtral-Mini-4B-Realtime-2602"
            print(f"[*] Chargement Voxtral 4B Optimisé (4-bit)... Target: {repo_id}")
            self.processor = AutoProcessor.from_pretrained(repo_id)
            
            # On force 4-bit pour économiser la VRAM (T4 = 16GB, mais souvent saturé par diarization)
            quant_config = BitsAndBytesConfig(
                load_in_4bit=True,
                bnb_4bit_compute_dtype=torch.float16,
                bnb_4bit_quant_type="nf4",
                bnb_4bit_use_double_quant=True
            )
            
            self.model = VoxtralRealtimeForConditionalGeneration.from_pretrained(
                repo_id, 
                quantization_config=quant_config,
                device_map="auto", 
                attn_implementation="sdpa"
            )
            print("[*] Voxtral chargé et quantifié.")

        elif model_id == "whisper":
            from faster_whisper import WhisperModel
            self.model = WhisperModel("large-v3", device=device, compute_type="float16")

        elif model_id == "qwen":
            from transformers import Qwen2AudioForConditionalGeneration, AutoProcessor, BitsAndBytesConfig
            repo_id = "Qwen/Qwen2-Audio-7B-Instruct"
            self.processor = AutoProcessor.from_pretrained(repo_id)
            quant_config = BitsAndBytesConfig(load_in_4bit=True, bnb_4bit_compute_dtype=torch_dtype)
            self.model = Qwen2AudioForConditionalGeneration.from_pretrained(
                repo_id, quantization_config=quant_config, device_map="auto", attn_implementation="sdpa"
            )

        # Diarization (Pyannote) — common to all jobs
        self.diarization = None
        self.live_buffer = [] # Pour la diarisation live
        self.asr_buffer = []  # Pour le contexte ASR live (sliding window)
        self.full_audio = []  # Pour la repasse finale (H-Quality)
        self.last_text = ""   # Pour le delta logic
        hf_token = os.environ.get("HF_TOKEN")
        if hf_token:
            from pyannote.audio import Pipeline, Model
            try:
                # Batch Diarization
                self.diarization = Pipeline.from_pretrained("pyannote/speaker-diarization-3.1", token=hf_token)
                self.diarization.to(torch.device(device))
                # Live Diarization (Fast Segmentation 3.0)
                self.live_diar = Model.from_pretrained("pyannote/segmentation-3.0", token=hf_token)
                self.live_diar.to(torch.device(device))
                print("[*] Diarisation SOTA (Batch & Live) activée.")
            except Exception as e:
                print(f"[!] Erreur Diarisation : {e}")
        else:
            print("[!] HF_TOKEN manquant : Diarisation désactivée.")

        if model_id not in ["voxtral", "whisper", "qwen"]:
            print(f"[*] Modèle '{model_id}' : mode mock.")

    # ---------- LIVE ----------
    async def transcribe_live_chunk(self, audio_bytes: bytes) -> dict:
        if self.model_id != "voxtral" or not hasattr(self, "model"):
            return {"text": "..."}

        import numpy as np
        audio = np.frombuffer(audio_bytes, dtype=np.int16).astype(np.float32) / 32768.0
        
        # 1. ASR (Voxtral Realtime) - CONTEXTUAL SLIDING WINDOW
        self.asr_buffer.append(audio)
        # On garde 2 secondes de contexte pour la stabilité (32000 samples)
        total_asr = sum(len(c) for c in self.asr_buffer)
        while total_asr > 32000 and len(self.asr_buffer) > 1:
            total_asr -= len(self.asr_buffer.pop(0))
        
        context_audio = np.concatenate(self.asr_buffer).flatten()
        inputs = self.processor(context_audio, sampling_rate=16000, return_tensors="pt")
        inputs = {k: v.to(self.model.device, dtype=self.model.dtype) if v.is_floating_point() else v.to(self.model.device) for k, v in inputs.items()}
        
        with torch.no_grad():
            outputs = self.model.generate(**inputs, max_new_tokens=64)
            full_text = self.processor.batch_decode(outputs, skip_special_tokens=True)[0].strip()

        # Debug
        if full_text:
            print(f"[Live ASR] raw: {full_text}")

        # Delta Logic : on renvoie le texte si c'est nouveau ou si ça a changé
        display_text = full_text
        if self.last_text and full_text.startswith(self.last_text):
            display_text = full_text[len(self.last_text):].strip()
        
        # Stabilité : on ne "valide" le texte que s'il est assez long
        if len(full_text) > 40:
            self.last_text = full_text[:-15] # On garde une marge d'incertitude
        elif not full_text:
            self.last_text = "" # Reset si silence

        # 2. Live Diarization (Background)
        speaker = "..."
        if hasattr(self, "live_diar"):
            self.live_buffer.append(audio)
            total_diar = sum(len(c) for c in self.live_buffer)
            while total_diar > 320000 and len(self.live_buffer) > 1:
                total_diar -= len(self.live_buffer.pop(0))

            async def _get_speaker():
                if total_diar < 16000: return "Speaker"
                # S'assurer que le buffer est bien 1D
                wav_data = np.concatenate(self.live_buffer)
                wav = torch.from_numpy(wav_data[None, :]).to(torch.device(device))
                with torch.no_grad():
                    seg = self.live_diar(wav)
                    spk_idx = torch.argmax(seg[0, -1, :]).item()
                    return f"Speaker {spk_idx}"
            
            speaker = await _get_speaker()
        
        # 3. Save for Final Pass
        self.full_audio.append(audio)

        return {"text": display_text, "speaker": speaker}

    async def finalize_live_session(self, client_id: str):
        """Perform a second high-quality pass on the full live stream."""
        if not self.full_audio: return
        
        import numpy as np, datetime, json
        print(f"[*] Fin de session Live pour {client_id} — Repasse haute qualité...")
        
        # On concatène tout l'audio capturé
        audio_full = np.concatenate(self.full_audio)
        self.full_audio = [] # Reset
        self.asr_buffer = []
        self.last_text  = ""

        # Sauvegarde temporaire pour process_batch (on réutilise la logique existante)
        temp_path = os.path.join(tempfile.gettempdir(), f"live_refine_{client_id}_{int(time.time())}.wav")
        import soundfile as sf
        sf.write(temp_path, audio_full, 16000)
        
        try:
            # On utilise Faster-Whisper pour la repasse car c'est le plus fiable
            # même si le live était en Voxtral.
            old_model = self.model_id
            if self.model_id != "whisper":
                # On switch temporairement si besoin (ou on reste sur le modèle actuel)
                pass 
                
            result = await self.process_batch(temp_path, f"live_{int(time.time())}")
            
            # Sauvegarde finale
            out_dir = os.path.abspath(os.path.join("transcriptions_terminees", client_id))

        finally:

    # ---------- BATCH ----------
    async def process_batch(self, audio_path: str, file_id: str = None) -> dict:
        print(f"[*] Inférence batch sur {audio_path}")
        loop = asyncio.get_event_loop()
        
        def _update_progress(msg: str):
            if file_id and file_id in jobs_db:
                jobs_db[file_id]["status"] = f"processing: {msg}"

        def _run():
            import ffmpeg, numpy as np
            # On utilise 16kHz comme standard pour VAD et Diarization
            sr_target = 16000
            
            # 1. Décodage total
            _update_progress("Décodage audio...")
            out, _ = (
                ffmpeg.input(audio_path)
                .output("pipe:", format="f32le", acodec="pcm_f32le", ac=1, ar=sr_target)
                .run(capture_stdout=True, capture_stderr=True, quiet=True)
            )
            y = np.frombuffer(out, dtype=np.float32).copy()
            
            # 2. Diarisation (Pyannote)
            diar_segments = []
            if self.diarization:
                _update_progress("Diarisation en cours...")
                # Pyannote attend un tensor [channels, samples]
                audio_tensor = torch.from_numpy(y[None, :])
                diarization = self.diarization({"waveform": audio_tensor, "sample_rate": sr_target})
                for turn, _, speaker in diarization.itertracks(yield_label=True):
                    diar_segments.append({"start": turn.start, "end": turn.end, "speaker": speaker})
                print(f"[*] Diarisation terminée : {len(diar_segments)} segments.")
            
            # 3. Transcription (VAD + Model)
            _update_progress("VAD & Inférence...")
            model_vad, utils = torch.hub.load(repo_or_dir='snakers4/silero-vad', model='silero_vad', trust_repo=True)
            (get_speech_timestamps, _, _, _, _) = utils
            speech_timestamps = get_speech_timestamps(torch.from_numpy(y), model_vad, sampling_rate=sr_target)
            
            # Construction du texte segment par segment
            full_transcript = []
            
            if self.model_id == "whisper":
                # Faster-whisper est excellent pour le batch long
                segments, _ = self.model.transcribe(audio_path, beam_size=5, language="fr")
                for s in segments:
                    # Trouver le speaker le plus présent sur ce segment
                    spk = "UNKNOWN"
                    if diar_segments:
                        overlap = 0
                        for ds in diar_segments:
                            ov = min(s.end, ds["end"]) - max(s.start, ds["start"])
                            if ov > overlap:
                                overlap = ov
                                spk = ds["speaker"]
                    
                    time_str = f"[{int(s.start//60):02d}:{int(s.start%60):02d}]"
                    line = f"{time_str} {spk}: {s.text.strip()}"
                    full_transcript.append(line)
            else:
                # Mode Chunking pour Voxtral / Qwen (plus gourmands)
                for i, ts in enumerate(speech_timestamps):
                    start_s, end_s = ts['start']/sr_target, ts['end']/sr_target
                    chunk = y[ts['start']:ts['end']]
                    pad = np.zeros(int(0.2 * sr_target), dtype=np.float32)
                    chunk_padded = np.concatenate([pad, chunk, pad])
                    
                    torch.cuda.empty_cache()
                    if self.model_id == "voxtral":
                        inputs = self.processor(chunk_padded, sampling_rate=sr_target, return_tensors="pt")
                        inputs = {k:v.to(self.model.device) for k,v in inputs.items()}
                        with torch.no_grad():
                            out = self.model.generate(**inputs, max_new_tokens=256)
                        part = self.processor.batch_decode(out, skip_special_tokens=True)[0]
                    else: # Qwen
                        prompt = "<|audio_only|>Transcription détaillée en français :"
                        inputs = self.processor(text=prompt, audios=chunk_padded, sampling_rate=sr_target, return_tensors="pt")
                        inputs = {k:v.to(self.model.device) for k,v in inputs.items()}
                        with torch.no_grad():
                            out = self.model.generate(**inputs, max_new_tokens=256)
                        part = self.processor.batch_decode(out, skip_special_tokens=True)[0].split(prompt)[-1]
                    
                    # Trouver speaker
                    spk = "UNKNOWN"
                    if diar_segments:
                        overlap = 0
                        for ds in diar_segments:
                            ov = min(end_s, ds["end"]) - max(start_s, ds["start"])
                            if ov > overlap: overlap = ov; spk = ds["speaker"]
                    
                    time_str = f"[{int(start_s//60):02d}:{int(start_s%60):02d}]"
                    full_transcript.append(f"{time_str} {spk}: {part.strip()}")
            
            return "\n".join(full_transcript)

        text = await loop.run_in_executor(None, _run)
        return {"text": text}

    return {"status": "ok"}

def _all_chunks_present(tmp_dir: str, total: int) -> bool:
    return all(os.path.exists(os.path.join(tmp_dir, f"chunk_{i}")) for i in range(total))


@app.post("/batch_chunk")
async def batch_chunk(
    background_tasks: BackgroundTasks,
    file_id: str = Form(...),
    client_id: str = Form("anonymous"),
    chunk_index: int = Form(...),
    total_chunks: int = Form(...),
    file: UploadFile = File(...),
):
    tmp_dir    = os.path.join(tempfile.gettempdir(), "asr_uploads", file_id)
    chunk_path = os.path.join(tmp_dir, f"chunk_{chunk_index}")
    os.makedirs(tmp_dir, exist_ok=True)

    with open(chunk_path, "wb") as f:
        f.write(await file.read())

    # Tous les chunks reçus → réassemblage + lancement inférence
    if chunk_index == total_chunks - 1:
        await asyncio.sleep(0.5)  # laisse le FS se synchroniser

        if not _all_chunks_present(tmp_dir, total_chunks):
            return JSONResponse(
                status_code=400,
                content={"error": "Certains chunks sont manquants — réessayez."},
            )

        final_path = os.path.join(tempfile.gettempdir(), f"{file_id}_{file.filename}")
        try:
            with open(final_path, "wb") as out:
                for i in range(total_chunks):
                    part = os.path.join(tmp_dir, f"chunk_{i}")
                    with open(part, "rb") as p:
                        out.write(p.read())
                    os.remove(part)
            os.rmdir(tmp_dir)
        except Exception as e:
            return JSONResponse(status_code=500, content={"error": f"Réassemblage échoué : {e}"})

        jobs_db[file_id] = {"status": "processing", "client_id": client_id}
        background_tasks.add_task(_gpu_job, final_path, file_id)
        return JSONResponse(content={"file_id": file_id, "status": "processing_started"})

    return JSONResponse(content={"file_id": file_id, "status": f"chunk_{chunk_index}_ok"})


@app.get("/status/{file_id}")

    return JSONResponse(content=jobs_db.get(file_id, {"status": "not_found"}))


async def _gpu_job(file_path: str, file_id: str):
    import json, datetime
    try:
        # Récupération du client_id stocké à l'upload
        client_id = jobs_db.get(file_id, {}).get("client_id", "anonymous")
        
        result = await asr_engine.process_batch(file_path, file_id)
        jobs_db[file_id] = {"status": "done", "result": result, "client_id": client_id}
        
        # Sauvegarde persistante isolée par utilisateur
        out_dir = os.path.abspath(os.path.join("transcriptions_terminees", client_id))
        os.makedirs(out_dir, exist_ok=True)
        
        # Tentative d'extraire le nom original (format: uuid_nom.m4a)
        orig_name = os.path.basename(file_path)
        if f"{file_id}_" in orig_name:
            orig_name = orig_name.split(f"{file_id}_", 1)[-1]
            
        safe_name = "".join(c for c in orig_name if c.isalnum() or c in ".-_")
        timestamp = datetime.datetime.now().strftime("%Y%m%d_%H%M%S")
        base_path = os.path.join(out_dir, f"{timestamp}_{safe_name}")
        
        # JSON complet
        with open(f"{base_path}.json", "w", encoding="utf-8") as f:
            json.dump({
                "file": orig_name,
                "id": file_id,
                "timestamp": timestamp,
                "result": result
            }, f, indent=2, ensure_ascii=False)
            
        # Texte brut
        with open(f"{base_path}.txt", "w", encoding="utf-8") as f:
            f.write(result.get("text", ""))
            
        print(f"[*] ✅ Sauvegarde locale terminée : {base_path}.[json|txt]")
        
    except Exception as e:
        jobs_db[file_id] = {"status": "error", "error": str(e)}
    finally:
        if os.path.exists(file_path):
            os.remove(file_path)
